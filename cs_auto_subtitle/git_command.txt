git add . //暂存文件
git commit -m "备注文本" //暂存添加至仓库
git log //查看提交信息 **commit id
	--stat //每次提交时修改的文件
	
git diff [commit id] // 查看某次提交内容

git reset --hard [commit id] //代码回溯
git checkout [commit id] //代码回溯

git branch  //当前有哪些分支 一般维持master的稳定性，develop用于开发，debug用于测试
git checkout -b develop //创建develop分支
git checkout master //切换至master分支
git merge develop //合并develop至当前分支
